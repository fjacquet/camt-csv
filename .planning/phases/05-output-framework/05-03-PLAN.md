```go
// Get formatter registry from container
registry := container.GetFormatterRegistry()
formatter, err := registry.Get(format)

    logger.Fatalf("Invalid format '%s': %v. Valid formats: standard, icompta", format, err)
}

delimiter := formatter.Delimiter()
logger.WithField("format", format).WithField("delimiter", string(delimiter)).Info("Using output format")
```
