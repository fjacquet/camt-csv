```go
// BatchConvert converts all Revolut investment CSV files in inputDir to standard CSV format in outputDir.
// Returns the count of successfully converted files.
func (a *Adapter) BatchConvert(ctx context.Context, inputDir, outputDir string) (int, error) {
    if a.logger == nil {
        a.logger = logging.NewLogrusAdapter("info", "text")
    }

    a.logger.Info("Starting batch conversion",
        logging.Field{Key: "inputDir", Value: inputDir},
        logging.Field{Key: "outputDir", Value: outputDir})

    // Ensure output directory exists
    if err := os.MkdirAll(outputDir, 0750); err != nil {
        return 0, fmt.Errorf("failed to create output directory: %w", err)
    }

    // Read input directory
    files, err := os.ReadDir(inputDir)
    if err != nil {
        return 0, fmt.Errorf("failed to read input directory: %w", err)
    }

    count := 0
    for _, file := range files {
        if file.IsDir() {
            continue
        }

        // Only process CSV files
        if !strings.HasSuffix(strings.ToLower(file.Name()), ".csv") {
            a.logger.Debug("Skipping non-CSV file", logging.Field{Key: "file", Value: file.Name()})
            continue
        }

        inputPath := filepath.Join(inputDir, file.Name())
        outputPath := filepath.Join(outputDir, file.Name())

        a.logger.Info("Converting file",
            logging.Field{Key: "input", Value: inputPath},
            logging.Field{Key: "output", Value: outputPath})

        // Validate format before conversion
        valid, err := a.ValidateFormat(inputPath)
        if err != nil || !valid {
            a.logger.WithError(err).Warn("Skipping invalid file", logging.Field{Key: "file", Value: file.Name()})
            continue
        }

        // Convert file
        if err := a.ConvertToCSV(ctx, inputPath, outputPath); err != nil {
            a.logger.WithError(err).Warn("Failed to convert file", logging.Field{Key: "file", Value: file.Name()})
            continue
        }

        count++
    }

    a.logger.Info("Batch conversion complete",
        logging.Field{Key: "filesConverted", Value: count})

    return count, nil
}
```

**Implementation notes:**
- Follows parser.BatchConverter interface signature
- Uses adapter's logger for consistent logging
- Validates each file before conversion (skip invalid files)
- Continues on individual file errors (don't fail entire batch)
- Returns count of successfully converted files
- Creates output directory if missing
