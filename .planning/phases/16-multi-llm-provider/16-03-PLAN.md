**Part A — Add SetEmbeddingClient to categorizer.go**

In `internal/categorizer/categorizer.go`, add the following method after the existing `SetStagingStore` method (around line 409):

```go
// SetEmbeddingClient replaces the AIClient used by the SemanticStrategy with the
// provided client. Call this after NewCategorizer when the embedding provider
// differs from the chat AI provider (e.g. OpenRouter for chat, Gemini for embeddings).
// Pass nil to disable the semantic tier.
func (c *Categorizer) SetEmbeddingClient(client AIClient) {
    for _, strategy := range c.strategies {
        if sem, ok := strategy.(*SemanticStrategy); ok {
            sem.client = client
            // Re-trigger background embedding init if a real client is now available
            if client != nil && !sem.initialized {
                go sem.initializeEmbeddings(context.Background(), c.categories)
            }
            return
        }
    }
}
```

Note: `sem.client` is an exported-within-package field (lowercase `client` in the same `categorizer` package), so this direct assignment is valid. Add `"context"` to categorizer.go imports if not already present (it is already imported).

2. Replace the AI client creation block (lines 82-88) with:

   ```go
   var chatClient categorizer.AIClient
   var embeddingClient categorizer.AIClient

   if cfg.AI.Enabled && cfg.AI.APIKey != "" {
       switch cfg.AI.Provider {
       case "openrouter":
           chatClient = categorizer.NewOpenRouterClient(
               logger,
               cfg.AI.RequestsPerMinute,
               cfg.AI.Model,
               cfg.AI.TimeoutSeconds,
               cfg.AI.APIKey,
               cfg.AI.BaseURL,
           )
           logger.WithFields(
               logging.Field{Key: "provider", Value: "openrouter"},
               logging.Field{Key: "model", Value: cfg.AI.Model},
           ).Info("AI provider: openrouter")

           // Embedding client: use Gemini if GEMINI_API_KEY is available
           geminiKey := os.Getenv("GEMINI_API_KEY")
           if geminiKey != "" {
               embeddingClient = categorizer.NewGeminiClient(logger, cfg.AI.RequestsPerMinute, "gemini-embedding-001", cfg.AI.TimeoutSeconds, geminiKey)
               logger.Info("Semantic tier: active (Gemini embeddings)")
           } else {
               logger.Info("Semantic tier: skipped (no embedding provider available)")
           }

       default: // "gemini" or any unrecognized value falls back to Gemini
           chatClient = categorizer.NewGeminiClient(logger, cfg.AI.RequestsPerMinute, cfg.AI.Model, cfg.AI.TimeoutSeconds, cfg.AI.APIKey)
           embeddingClient = chatClient
           logger.WithFields(
               logging.Field{Key: "provider", Value: "gemini"},
               logging.Field{Key: "model", Value: cfg.AI.Model},
           ).Info("AI provider: gemini")
           logger.Info("Semantic tier: active (Gemini embeddings)")
       }
   } else {
       logger.Info("AI categorization disabled")
   }
   ```
